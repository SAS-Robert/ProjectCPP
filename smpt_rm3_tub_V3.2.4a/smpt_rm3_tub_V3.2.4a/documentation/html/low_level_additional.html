typedef enum
{
    Smpt_Length_Number_Of_Switches = 512, // Array length of number of switches
    Smpt_Length_Demux_Config       = Smpt_Length_Number_Of_Switches / 8, // Array length of demux config
    Smpt_Length_Demux_Id           = 50   //  Array length of demux id

} Smpt_Ll_Length;

typedef struct
{
    bool enable_demux;              //   False - The demultiplexer is disabled,
                                    //    True - The demultiplexer is enabled
} Smpt_ll_init;

typedef struct
{
    char demux_id[Smpt_Length_Demux_Id];  // Type ID of the demultiplexer

} Smpt_ll_init_ack;

typedef struct
{
    bool modify_demux;         //    Value | Description
                               //    ----- | -----------
                               //    False | Do not transfer the demux configuration
                               //    True  | Transfer and use the demux configuration
    uint8_t demux_config[Smpt_Length_Demux_Config]; // Demux for configuration for all switches.
                                                    // Should not be written directly. Please use
                                                    // smpt_ll_demux_convert()
    uint8_t demux_length; // Number of bytes for demux_config [0 - 64]

} Smpt_ll_channel_config;  // \struct Smpt_ll_channel_config
                           //  Struct for the Ll_channel_config command.
                           //  \see smpt_send_ll_channel_config()

typedef enum
{
    Smpt_Ll_Demux_Unused   = 0,  // Electrode is not used
    Smpt_Ll_Demux_Positive = 1,  // Electrode is positive
    Smpt_Ll_Demux_Negative = 2   // Electrode is negative
} Smpt_Ll_Demux_Element_Mode; // \enum Smpt_Ll_Demux_Element_Mode Mode of the demux electrode
                              // \see Smpt_ll_demux

typedef struct
{
    Smpt_Demux demux_id; // Type ID of the demultiplexer
    Smpt_Ll_Demux_Element_Mode electrodes[Smpt_Length_Number_Of_Switches]; // electrodes configuration
} Smpt_ll_demux;  //  \struct Smpt_ll_demux
                  //  Struct for the demultiplexer configuration
                  //  \see smpt_get_ll_channel_config_ack()

typedef struct
{
    // KS1 (Short circuit switch near electrodes)
    // KS2 (Short circuit switch near EMG-Chip)
    // LS1 (Connecting switch (Laengsschalter))
    uint8_t ks_1_start;     // [0 .. 1 .. 255] -> [0 .. 0.125 .. 31.875] ms
    uint8_t ks_1_duration;  // [0 .. 1 .. 255] -> [0 .. 0.125 .. 31.875] ms
    uint8_t ks_2_start;     // [0 .. 1 .. 255] -> [0 .. 0.125 .. 31.875] ms
    uint8_t ks_2_duration;  // [0 .. 1 .. 255] -> [0 .. 0.125 .. 31.875] ms
    uint8_t ls_stop;        // [0 .. 1 .. 255] -> [0 .. 0.125 .. 31.875] ms
    uint8_t packet_number;  // [0 - 63] Packet number
} Smpt_ll_emg_switches;  // \struct Smpt_ll_emg_switches
                         //  Struct for the Ll_emg_switches command. The command is not required
                         //  for normal operation and should only be used if you know what you are
                         //  doing.
                         //  \see smpt_send_ll_emg_switches()
